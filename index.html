<script>

    document.getElementById('form-punteggio').addEventListener('submit', function(e) {

        e.preventDefault();


        // Ottenere i valori dal form

        const eta = document.getElementById('eta').value;

        const eziologia = document.getElementById('eziologia').value;

        const shunt = document.getElementById('shunt').value;


        let punteggio = 0;


        // Logica di calcolo del punteggio

        if (eta === "0-1") {

            if (shunt === "yes") {

                punteggio = 0;

            } else {

                punteggio = 10;

            }

        } else if (eta === "1-6") {

            if (eziologia === "myelomeningocele") {

                punteggio = 20;

            } else if (shunt === "no") {

                punteggio = 10;

            }

        } else if (eta === "6-12") {

            if (eziologia === "aqueductal") {

                punteggio = 30;

            }

        } else if (eta === "1-10") {

            punteggio = 40;

        } else if (eta === ">10") {

            punteggio = 50;

        }


        // Mostra il risultato

        document.getElementById('risultato').innerText = 'Il punteggio calcolato è: ' + punteggio;

    });

</script>
